303 template<bool iomode, IOPad pad>
304 void writeFlag(bool flag, std::ostream &os){
305  if (iomode == mode_ascii){
306  os << ((flag) ? "1" : "0");
307  if ((pad == pad_rspace) || ((pad == pad_auto) && flag)) os << " ";
308  if ((pad == pad_line) || ((pad == pad_auto) && !flag)) os << std::endl;
309  }else{
310  char cflag = ((flag) ? 'y' : 'n');
311  os.write(&cflag, sizeof(char));
312  }
313 }

323  os >> flag;
324  return (flag != 0);

328  return (cflag == 'y');

336 template<bool iomode, IOPad pad, typename VecType>
337 void writeVector(const std::vector<VecType> &x, std::ostream &os){
338  if (iomode == mode_ascii){
339  if (pad == pad_lspace)
340  for(auto i : x) os << " " << i;
341  if (pad == pad_rspace)
342  for(auto i : x) os << i << " ";
343  if ((pad == pad_none) || (pad == pad_line)){
344  os << x[0];
345  for(size_t i = 1; i < x.size(); i++) os << " " << x[i];
346  if (pad == pad_line) os << std::endl;
347  }
348  }else{
349  os.write((char*) x.data(), x.size() * sizeof(VecType));
350  }
351 }
