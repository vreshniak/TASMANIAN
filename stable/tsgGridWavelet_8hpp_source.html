143 template<> struct GridReaderVersion5<GridWavelet>{
144  template<typename iomode> static std::unique_ptr<GridWavelet> read(AccelerationContext const *acc, std::istream &is){
145  std::unique_ptr<GridWavelet> grid = Utils::make_unique<GridWavelet>(acc);
146 
147  grid->num_dimensions = IO::readNumber<iomode, int>(is);
148  grid->num_outputs = IO::readNumber<iomode, int>(is);
149  grid->order = IO::readNumber<iomode, int>(is);
150  grid->rule1D.updateOrder(grid->order);
151 
152  if (IO::readFlag<iomode>(is)) grid->points = MultiIndexSet(is, iomode());
153  if (std::is_same<iomode, IO::mode_ascii_type>::value){ // backwards compatible: surpluses and needed, or needed and surpluses
154  if (IO::readFlag<iomode>(is))
155  grid->coefficients = IO::readData2D<iomode, double>(is, grid->num_outputs, grid->points.getNumIndexes());
156  if (IO::readFlag<iomode>(is)) grid->needed = MultiIndexSet(is, iomode());
157  }else{
158  if (IO::readFlag<iomode>(is)) grid->needed = MultiIndexSet(is, iomode());
159  if (IO::readFlag<iomode>(is))
160  grid->coefficients = IO::readData2D<iomode, double>(is, grid->num_outputs, grid->points.getNumIndexes());
161  }
162 
163  if (grid->num_outputs > 0) grid->values = StorageSet(is, iomode());
164  grid->buildInterpolationMatrix();
165 
166  return grid;
167  }
