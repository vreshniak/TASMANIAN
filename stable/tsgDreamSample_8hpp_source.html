263 template<TypeSamplingForm form = regform>
264 DreamPDF posterior(DreamModel model,
265  DreamLikelihood likelihood,
266  DreamPrior prior){
267  return [=](const std::vector<double> &candidates, std::vector<double> &values)->void{
268  std::vector<double> model_outs;
269  model(candidates, model_outs);
270  likelihood(form, model_outs, values);
271 
272  std::vector<double> prior_vals(values.size());
273  prior(form, candidates, prior_vals);
274 
275  auto iv = values.begin();
276  if (form == regform){
277  for(auto p : prior_vals) *iv++ *= p;
278  }else{
279  for(auto p : prior_vals) *iv++ += p;
280  }
281  };
282 }

417 template<TypeSamplingForm form = regform>
418 void SampleDREAM(int num_burnup, int num_collect,
419  DreamPDF probability_distribution,
420  DreamDomain inside,
421  TasmanianDREAM &state,
422  std::function<void(std::vector<double> &x)> independent_update = no_update,
423  std::function<double(void)> differential_update = const_one,
424  std::function<double(void)> get_random01 = tsgCoreUniform01){
425 
426  size_t num_chains = (size_t) state.getNumChains(), num_dimensions = (size_t) state.getNumDimensions();
427  double unitlength = (double) num_chains;
428 
429  if (num_chains == 0) return; // no sampling with a null state
430 
431  if (!state.isStateReady()) throw std::runtime_error("ERROR: DREAM sampling requires that the setState() has been called first on the TasmanianDREAM.");
432 
433  if (!state.isPDFReady()) // initialize probability density (if not initialized already)
434  state.setPDFvalues(probability_distribution);
435 
436  if (num_collect > 0) // pre-allocate memory for the new history
437  state.expandHistory(num_collect);
438 
439  int total_iterations = std::max(num_burnup, 0) + std::max(num_collect, 0);
440  for(int t = 0; t < total_iterations; t++){
441  std::vector<double> candidates, values;
442  candidates.reserve(num_chains * num_dimensions);
443  values.reserve(num_chains);
444 
445  std::vector<bool> valid(num_chains, true); // keep track whether the samples need to be evaluated
446 
447  for(size_t i=0; i<num_chains; i++){
448  std::vector<double> propose(num_dimensions);
449 
450  size_t jindex = (size_t) (get_random01() * unitlength);
451  size_t kindex = (size_t) (get_random01() * unitlength);
452  if (jindex >= num_chains) jindex = num_chains - 1; // this is needed in case get_random01() returns 1
453  if (kindex >= num_chains) jindex = num_chains - 1;
454 
455  state.getIJKdelta(i, jindex, kindex, differential_update(), propose); // propose = s_i + w ( s_k - s_j)
456  independent_update(propose); // propose += correction
457 
458  if (inside(propose)){
459  candidates.insert(candidates.end(), propose.begin(), propose.end());
460  values.resize(values.size() + 1);
461  }else{
462  valid[i] = false;
463  }
464  }
465 
466  if (!candidates.empty()) // block the pathological case of all proposals leaving the domain
467  probability_distribution(candidates, values);
468 
469  std::vector<double> new_state(num_chains * num_dimensions), new_values(num_chains);
470 
471  auto icand = candidates.begin(); // loop over all candidates and values, accept or reject
472  auto ival = values.begin();
473 
474  size_t accepted = 0;
475 
476  for(size_t i=0; i<num_chains; i++){
477  bool keep_new = valid[i]; // if not valid, automatically reject
478  if (valid[i]){ // apply random test
479  if (*ival > state.getPDFvalue(i)){ // if the new value has higher probability, automatically accept
480  keep_new = true;
481  }else{
482  if (form == regform){
483  keep_new = (*ival / state.getPDFvalue(i) >= get_random01()); // keep if the new value has higher probability
484  }else{
485  keep_new = (*ival - state.getPDFvalue(i) >= log(get_random01()));
486  }
487  //std::cout << "Trsh = " << *ival / state.getPDFvalue(i) << " " << ((keep_new) ? "Accept" : "Reject") << std:: endl;
488  }
489  }
490 
491  if (keep_new){
492  std::copy_n(icand, num_dimensions, new_state.begin() + i * num_dimensions);
493  new_values[i] = *ival;
494  accepted++; // accepted one more proposal
495  }else{ // reject and reuse the old state
496  state.getChainState((int) i, &*(new_state.begin() + i * num_dimensions));
497  new_values[i] = state.getPDFvalue(i);
498  }
499 
500  if (valid[i]){ // kept or rejected, if this sample was valid then move to the next sample in the list
501  std::advance(icand, num_dimensions);
502  ival++;
503  }
504  }
505 
506  state.setState(new_state);
507  state.setPDFvalues(new_values);
508 
509  if (t >= num_burnup)
510  state.saveStateHistory(accepted);
511  }
512 }

524 template<TypeSamplingForm form = regform>
525 void SampleDREAM(int num_burnup, int num_collect,
526  DreamPDF probability_distribution,
527  DreamDomain inside,
528  TasmanianDREAM &state,
529  TypeDistribution dist, double magnitude,
530  std::function<double(void)> differential_update = const_one,
531  std::function<double(void)> get_random01 = tsgCoreUniform01){
532  if (dist == dist_uniform){
533  SampleDREAM<form>(num_burnup, num_collect, probability_distribution, inside, state,
534  [&](std::vector<double> &x)->void{ applyUniformUpdate(x, magnitude, get_random01); }, differential_update, get_random01);
535  }else if (dist == dist_gaussian){
536  SampleDREAM<form>(num_burnup, num_collect, probability_distribution, inside, state,
537  [&](std::vector<double> &x)->void{ applyGaussianUpdate(x, magnitude, get_random01); }, differential_update, get_random01);
538  }else{ // assuming none
539  SampleDREAM<form>(num_burnup, num_collect, probability_distribution, inside, state, no_update, differential_update, get_random01);
540  }
541 }
