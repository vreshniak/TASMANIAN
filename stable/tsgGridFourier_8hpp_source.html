123  template<typename T, bool interwoven>
124  void computeBasis(const MultiIndexSet &work, const T x[], T wreal[], T wimag[]) const{
125  int num_points = work.getNumIndexes();
126 
127  std::vector<std::vector<std::complex<T>>> cache(num_dimensions);
128  for(int j=0; j<num_dimensions; j++){
129  cache[j].resize(max_power[j] +1);
130  cache[j][0] = std::complex<T>(1.0, 0.0);
131 
132  T theta = -2.0 * Maths::pi * x[j];
133  std::complex<T> step(std::cos(theta), std::sin(theta));
134  std::complex<T> pw(1.0, 0.0);
135  for(int i=1; i<max_power[j]; i += 2){
136  pw *= step;
137  cache[j][i] = pw;
138  cache[j][i+1] = std::conj<T>(pw);
139  }
140  }
141 
142  for(int i=0; i<num_points; i++){
143  const int *p = work.getIndex(i);
144 
145  std::complex<T> v(1.0, 0.0);
146  for(int j=0; j<num_dimensions; j++){
147  v *= cache[j][p[j]];
148  }
149 
150  if (interwoven){
151  wreal[2*i] = v.real();
152  wreal[2*i+1] = v.imag();
153  }else{
154  wreal[i] = v.real();
155  wimag[i] = v.imag();
156  }
157  }
158  }

218  grid->updated_tensors = MultiIndexSet(is, iomode());

221  grid->updated_active_tensors = MultiIndexSet(is, iomode());
