74 template<TypeOneDRule rule, bool isZeroOrder>
75 class templRuleLocalPolynomial : public BaseRuleLocalPolynomial{
76 public:
77  templRuleLocalPolynomial() = default;
78  ~templRuleLocalPolynomial() = default;
79 
80  int getMaxOrder() const override{ return max_order; }
81  void setMaxOrder(int order) override{ max_order = order; }
82 
83  TypeOneDRule getType() const override{ return rule; }
84 
85  int getNumPoints(int level) const override{
86  if (isZeroOrder){
87  int n = 1;
88  while (level-- > 0) n *= 3;
89  return n;
90  }else{
91  if ((rule == rule_localp) || (rule == rule_semilocalp)){
92  return (level == 0) ? 1 : ((1 << level) + 1);
93  }else if (rule == rule_localp0){
94  return (1 << (level+1)) -1;
95  }else{ // rule == rule_localpb
96  return ((1 << level) + 1);
97  }
98  }
99  }
100 
101  double getNode(int point) const override{
102  if (isZeroOrder){
103  return -2.0 + (1.0 / ((double) Maths::int3log3(point))) * (3*point + 2 - point % 2);
104  }else{
105  if ((rule == rule_localp) || (rule == rule_semilocalp)){
106  if (point == 0) return 0.0;
107  if (point == 1) return -1.0;
108  if (point == 2) return 1.0;
109  return ((double)(2*point - 1)) / ((double) Maths::int2log2(point - 1)) - 3.0;
110  }else if (rule == rule_localp0){
111  return ((double)(2*point +3) ) / ((double) Maths::int2log2(point + 1) ) - 3.0;
112  }else{ // rule == rule_localpb
113  if (point == 0) return -1.0;
114  if (point == 1) return 1.0;
115  if (point == 2) return 0.0;
116  return ((double)(2*point - 1)) / ((double) Maths::int2log2(point - 1)) - 3.0;
117  }
118  }
119  }
120 
121  int getLevel(int point) const override{
122  if (isZeroOrder){
123  int level = 0;
124  while(point >= 1){ point /= 3; level += 1; }
125  return level;
126  }else{
127  if ((rule == rule_localp) || (rule == rule_semilocalp)){
128  return (point == 0) ? 0 : (point == 1) ? 1 : (Maths::intlog2(point - 1) + 1);
129  }else if (rule == rule_localp0){
130  return Maths::intlog2(point + 1);
131  }else{ // rule == rule_localpb
132  return ((point == 0) || (point == 1)) ? 0 : (Maths::intlog2(point - 1) + 1);
133  }
134  }
135  }
136  double getSupport(int point) const override{
137  if (isZeroOrder){
138  return 1.0 / (double) Maths::int3log3(point);
139  }else{
140  if ((rule == rule_localp) || (rule == rule_semilocalp)){
141  return (point == 0) ? 1.0 : 1.0 / ((double) Maths::int2log2(point - 1));
142  }else if (rule == rule_localp0){
143  return 1.0 / ((double) Maths::int2log2(point + 1));
144  }else{ // rule == rule_localpb
145  return ((point == 0) || (point == 1)) ? 2.0 : 1.0 / ((double) Maths::int2log2(point - 1));
146  }
147  }
148  }
149 
150  int getMaxNumKids() const override{ return (isZeroOrder) ? 4 : 2; }
151  int getMaxNumParents() const override{ return ((isZeroOrder || (rule == rule_semilocalp) || (rule == rule_localpb)) ? 2 : 1); }
152 
153  int getParent(int point) const override{
154  if (isZeroOrder){
155  return (point == 0) ? -1 : point / 3;
156  }else{
157  if ((rule == rule_localp) || (rule == rule_semilocalp)){
158  int dad = (point + 1) / 2;
159  if (point < 4) dad--;
160  return dad;
161  }else if (rule == rule_localp0){
162  if (point == 0) return -1;
163  return (point - 1) / 2;
164  }else{ // rule == rule_localpb
165  return (point < 2) ? -1 : ((point + 1) / 2);
166  }
167  }
168  }
169 
170  int getStepParent(int point) const override{
171  if (isZeroOrder){
172  int i3l3 = Maths::int3log3(point);
173  if (point == i3l3/3) return -1;
174  if (point == i3l3-1) return -1;
175  if ((point % 3 == 2) && (point % 2 == 0)) return point / 3 + 1;
176  if ((point % 3 == 0) && (point % 2 == 1)) return point / 3 - 1;
177  return -1;
178  }else{
179  if (rule == rule_semilocalp){
180  if (point == 3) return 2;
181  if (point == 4) return 1;
182  }else if (rule == rule_localpb){
183  if (point == 2) return 0;
184  }
185  return -1;
186  }
187  }
188 
189  int getKid(int point, int kid_number) const override{
190  if (isZeroOrder){
191  if (point == 0) return (kid_number == 0) ? 1 : (kid_number==1) ? 2 : -1;
192  if (kid_number == 3){
193  int i3l3 = Maths::int3log3(point);
194  if (point == i3l3/3) return -1;
195  if (point == i3l3-1) return -1;
196  return (point % 2 == 0) ? 3*point + 3 : 3*point - 1;
197  }
198  return 3*point + kid_number;
199  }else if ((rule == rule_localp) || (rule == rule_semilocalp)){
200  if (kid_number == 0){
201  if (point == 0) return 1;
202  if (point == 1) return 3;
203  if (point == 2) return 4;
204  return 2*point-1;
205  }else{
206  if (point == 0) return 2;
207  if ((point == 1) || (point == 2)) return -1;
208  return 2*point;
209  }
210  }else if (rule == rule_localp0){
211  return 2*point + ((kid_number == 0) ? 1 : 2);
212  }else{ // localpb
213  if ((point == 0) || (point == 1)) return (kid_number == 0) ? 2 : -1;
214  return 2*point - ((kid_number == 0) ? 1 : 0);
215  }
216  }
217 
218  double evalRaw(int point, double x) const override{
219  if (isZeroOrder){
220  if (std::abs(x - getNode(point)) > getSupport(point)) return 0.0; // maybe can speed this up
221  return 1.0;
222  }else{
223  if ((rule == rule_localp) || (rule == rule_semilocalp)){
224  if (point == 0) return 1.0;
225  if (rule == rule_semilocalp){
226  if (point == 1) return 0.5 * x * (x - 1.0);
227  if (point == 2) return 0.5 * x * (x + 1.0);
228  }
229  }
230  double xn = scaleX(point, x);
231  if (rule != rule_semilocalp) if (max_order == 1) return 1.0 - std::abs(xn);
232  if (max_order == 2) return evalPWQuadratic(point, xn);
233  if (max_order == 3) return evalPWCubic(point, xn);
234  return evalPWPower(point, xn);
235  }
236  }
237 
238  double evalSupport(int point, double x, bool &isSupported) const override{
239  if (isZeroOrder){
240  double distance = std::abs(x - getNode(point)), support = getSupport(point); // can speed this up, see above
241  isSupported = (distance <= (2.0) * support);
242  return (distance > support) ? 0.0 : 1.0;
243  }else{
244  isSupported = true;
245  if ((rule == rule_localp) || (rule == rule_semilocalp)){
246  if (point == 0) return 1.0;
247  if (rule == rule_semilocalp){
248  if (point == 1) return 0.5 * x * (x - 1.0);
249  if (point == 2) return 0.5 * x * (x + 1.0);
250  }
251  }
252  double xn = scaleX(point, x);
253  if (std::abs(xn) <= 1.0){
254  if (rule != rule_semilocalp) if (max_order == 1) return 1.0 - std::abs(xn);
255  if (max_order == 2) return evalPWQuadratic(point, xn);
256  if (max_order == 3) return evalPWCubic(point, xn);
257  return evalPWPower(point, xn);
258  }else{
259  isSupported = false;
260  return 0.0;
261  }
262  }
263  }
264  double getArea(int point, std::vector<double> const &w, std::vector<double> const &x) const override{
265  if (isZeroOrder){
266  return 2.0 * getSupport(point);
267  }else{
268  if (rule == rule_localp){
269  if (point == 0) return 2.0;
270  if ((point == 1) || (point == 2)) return 0.5;
271  if (max_order == 1) return getSupport(point);
272  if ((max_order == 2) || (max_order == 3) || (point <= 8)) return (4.0/3.0) * getSupport(point);
273  }else if (rule == rule_semilocalp){
274  if (point == 0) return 2.0;
275  if ((point == 1) || (point == 2)) return 1.0/3.0;
276  if ((max_order == 2) || (max_order == 3) || (point <= 4)) return (4.0/3.0) * getSupport(point);
277  }else if (rule == rule_localpb){
278  if ((point == 0) || (point == 1)) return 1.0;
279  if (max_order == 1) return getSupport(point);
280  if ((max_order == 2) || (max_order == 3) || (point <= 4)) return (4.0/3.0) * getSupport(point);
281  }else if (rule == rule_localp0){
282  if (max_order == 1) return getSupport(point);
283  if ((max_order == 2) || (max_order == 3) || (point <= 2)) return (4.0/3.0) * getSupport(point);
284  }
285  double sum = 0.0;
286  for(size_t i=0; i<w.size(); i++) sum += w[i] * evalPWPower(point, x[i]);
287  return sum * getSupport(point);
288  }
289  }

291 protected:
292  double scaleX(int point, double x) const{
293  if (rule == rule_localp0){
294  if (point == 0) return x;
295  return ((double) Maths::int2log2(point + 1) * (x + 3.0) - 3.0 - (double) (2*point));
296  }else{
297  if (rule == rule_localp){
298  if (point == 0) return x;
299  if (point == 1) return (x + 1.0);
300  if (point == 2) return (x - 1.0);
301  }else if (rule == rule_localpb){
302  if (point == 0) return (x + 1.0) / 2.0;
303  if (point == 1) return (x - 1.0) / 2.0;
304  if (point == 2) return x;
305  }
306  return ((double) Maths::int2log2(point - 1) * (x + 3.0) + 1.0 - (double) (2*point));
307  }
308  }

310  double evalPWQuadratic(int point, double x) const{
311  if (rule == rule_localp){
312  if (point == 1) return 1.0 - x;
313  if (point == 2) return 1.0 + x;
314  }else if (rule == rule_localpb){
315  if (point == 0) return 1.0 - x;
316  if (point == 1) return 1.0 + x;
317  }
318  return (1.0 - x) * (1.0 + x);
319  }
320  double evalPWCubic(int point, double x) const{
321  if (rule == rule_localp){
322  if (point == 0) return 1.0;
323  if (point == 1) return 1.0 - x;
324  if (point == 2) return 1.0 + x;
325  if (point <= 4) return (1.0 - x) * (1.0 + x);
326  }else if (rule == rule_localpb){
327  if (point == 0) return 1.0 - x;
328  if (point == 1) return 1.0 + x;
329  if (point == 2) return (1.0 - x) * (1.0 + x);
330  }else if (rule == rule_localp0){
331  if (point == 0) return (1.0 - x) * (1.0 + x);
332  }
333  return (point % 2 == 0) ? (1.0 - x) * (1.0 + x) * (3.0 + x) / 3.0 : (1.0 - x) * (1.0 + x) * (3.0 - x) / 3.0;
334  }
335  double evalPWPower(int point, double x) const{
336  if (rule == rule_localp) if (point <= 8) return evalPWCubic(point, x); // if order is cubic or less, use the hard-coded functions
337  if (rule == rule_semilocalp) if (point <= 4) return evalPWCubic(point, x);
338  if (rule == rule_localpb) if (point <= 4) return evalPWCubic(point, x);
339  if (rule == rule_localp0) if (point <= 2) return evalPWCubic(point, x);
340  int level = getLevel(point);
341  int most_turns = 1;
342  double value = (1.0 - x)*(1.0 + x), phantom_distance = 1.0;
343  int max_ancestors; // maximum number of ancestors to consider, first set the possible max then constrain by max_order
344  if (rule == rule_localp) max_ancestors = level-2; // number of ancestors to consider (two counted in the initialization of value)
345  if (rule == rule_semilocalp) max_ancestors = level-1; // semi-localp and localpb add one more ancestor due to the global support at levels 1 and 0 (respectively)
346  if (rule == rule_localpb) max_ancestors = level-1;
347  if (rule == rule_localp0) max_ancestors = level; // localp0 adds two ancestors, the assumed zero nodes at the boundary
348  if (max_order > 0) max_ancestors = std::min(max_ancestors, max_order - 2); // use the minimum of the available ancestors or the order restriction
349 
350  for(int j=0; j < max_ancestors; j++){
351  // Lagrange polynomial needs to be constructed using normalized x (basis support (-1, 1)).
352  // The support of the basis is equal to 2, thus we use units of "half-support"
353  // The first two nodes are used in the initialization of value, those are the nearest ancestors (in spacial distance, not hierarchy level).
354  // The other nodes are "phantoms" that lay strictly outside of the support [-1, 1] (i.e., not on the edge)
355  // The walking distance more than doubles and is an odd number (due to the half-support)
356  // The walk through the ancestors can take a left or right turn at each step,
357  // most_turns is the total number of turns possible for the current ancestor, turns (or most_turns - 1 - turns) is the actual number of turns
358  // Every time we turn, we backtrack and we lose 2 units, the phantom node is at maximum distance minus 2 time the number of turns (to the left or right)
359  most_turns *= 2;
360  phantom_distance = 2.0 * phantom_distance + 1.0;
361  int turns = (rule == rule_localp0) ? ((point+1) % most_turns) : ((point-1) % most_turns);
362  double node = (turns < most_turns / 2) ? (phantom_distance - 2.0 * ((double) turns)) : (-phantom_distance + 2.0 * ((double) (most_turns - 1 - turns)));
363  value *= ( x - node ) / ( - node);
364  }
365  return value;
366  }

369 inline std::unique_ptr<BaseRuleLocalPolynomial> makeRuleLocalPolynomial(TypeOneDRule rule, int order){
370  if (order == 0){
371  return Utils::make_unique<templRuleLocalPolynomial<rule_localp, true>>();
372  }else{
373  std::unique_ptr<BaseRuleLocalPolynomial> rule_pntr = [=]()->std::unique_ptr<BaseRuleLocalPolynomial>{
374  if (rule == rule_localp){
375  return Utils::make_unique<templRuleLocalPolynomial<rule_localp, false>>();
376  }else if (rule == rule_semilocalp){
377  return Utils::make_unique<templRuleLocalPolynomial<rule_semilocalp, false>>();
378  }else if (rule == rule_localp0){
379  return Utils::make_unique<templRuleLocalPolynomial<rule_localp0, false>>();
380  }else{ // must be (rule == rule_localpb)
381  return Utils::make_unique<templRuleLocalPolynomial<rule_localpb, false>>();
382  }
383  }();
384  rule_pntr->setMaxOrder(order);
385  return rule_pntr;
386  }
