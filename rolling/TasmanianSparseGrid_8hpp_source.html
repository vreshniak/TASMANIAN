1748  DomainInsideSignature getDomainInside() const{
1749  auto rule = getRule();
1750  if ((rule == TasGrid::rule_gausshermite) || (rule == TasGrid::rule_gausshermiteodd)){ // unbounded domain
1751  return [](std::vector<double> const &)->bool{ return true; };
1752  }else if ((rule == TasGrid::rule_gausslaguerre) || (rule == TasGrid::rule_gausslaguerreodd)){ // bounded from below
1753  if (domain_transform_a.empty()){
1754  return [](std::vector<double> const &x)->bool{
1755  for(auto const &v : x) if (v < 0.0) return false;
1756  return true;
1757  };
1758  }else{
1759  size_t dims = (size_t) getNumDimensions();
1760  return [=](std::vector<double> const &x)->bool{
1761  for(size_t i=0; i<dims; i++) if (x[i] < domain_transform_a[i]) return false;
1762  return true;
1763  };
1764  }
1765  }else{
1766  if (domain_transform_a.empty()){
1767  if (isFourier()){
1768  return [](std::vector<double> const &x)->bool{
1769  for(auto const &v : x) if ((v < 0.0) || (v > 1.0)) return false;
1770  return true;
1771  };
1772  }else{
1773  return [](std::vector<double> const &x)->bool{
1774  for(auto const &v : x) if ((v < -1.0) || (v > 1.0)) return false;
1775  return true;
1776  };
1777  }
1778  }else{
1779  size_t dims = (size_t) getNumDimensions();
1780  return [=](std::vector<double> const &x)->bool{
1781  for(size_t i=0; i<dims; i++)
1782  if ((x[i] < domain_transform_a[i]) || (x[i] > domain_transform_b[i])) return false;
1783  return true;
1784  };
1785  }
1786  }
