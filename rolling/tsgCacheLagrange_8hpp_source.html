89  static void cacheLevel(int level, double x, const OneDimensionalWrapper &rule, T *cc){
90  const double *nodes = rule.getNodes(level);
91  const double *coeff = rule.getCoefficients(level);
92  int num_points = rule.getNumPoints(level);
93 
94  cc[0] = 1.0;
95  T c = 1.0;
96  for(int j=0; j<num_points-1; j++){
97  c *= (x - nodes[j]);
98  cc[j+1] = c;
99  }
100  c = (rule.getType() == rule_clenshawcurtis0) ? (x * x - 1.0) : 1.0;
101  cc[num_points-1] *= c * coeff[num_points-1];
102  for(int j=num_points-2; j>=0; j--){
103  c *= (x - nodes[j+1]);
104  cc[j] *= c * coeff[j];
105  }
106  }

109  T getLagrange(int dimension, int level, int local) const{
