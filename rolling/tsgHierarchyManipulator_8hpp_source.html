124 inline void touchAllImmediateRelatives(std::vector<int> &point, MultiIndexSet const &mset, BaseRuleLocalPolynomial const *rule, std::function<void(int i)> apply){
125  int max_kids = rule->getMaxNumKids();
126  for(auto &v : point){
127  int save = v; // replace one by one each index of p with either parent or kid
128 
129  // check the parents
130  v = rule->getParent(save);
131  if (v > -1){
132  int parent_index = mset.getSlot(point);
133  if (parent_index > -1)
134  apply(parent_index);
135  }
136 
137  v = rule->getStepParent(save);
138  if (v > -1){
139  int parent_index = mset.getSlot(point);
140  if (parent_index > -1)
141  apply(parent_index);
142  }
143 
144  for(int k=0; k<max_kids; k++){
145  v = rule->getKid(save, k);
146  if (v > -1){
147  int kid_index = mset.getSlot(point);
148  if (kid_index > -1)
149  apply(kid_index);
150  }
151  }
152 
153  v = save; // restore the original index for the next iteration
154  }
155 }

177 inline MultiIndexSet getLargestConnected(MultiIndexSet const &current, MultiIndexSet const &candidates, BaseRuleLocalPolynomial const *rule){
178  if (candidates.empty()) return MultiIndexSet();
179  auto num_dimensions = candidates.getNumDimensions();
180 
181  // always consider the points without parents
182  MultiIndexSet level_zero = getLevelZeroPoints(num_dimensions, rule);
183 
184  MultiIndexSet result; // keep track of the cumulative result
185  MultiIndexSet total = current; // forms a working copy of the entire merged graph
186 
187  // do not consider the points already included in total, complexity is level_zero.getNumIndexes()
188  if (!total.empty()) level_zero = level_zero - total;
189 
190  if (level_zero.getNumIndexes() > 0){ // level zero nodes are missing from current
191  Data2D<int> roots(num_dimensions, 0);
192  for(int i=0; i<level_zero.getNumIndexes(); i++){
193  std::vector<int> p(level_zero.getIndex(i), level_zero.getIndex(i) + num_dimensions);
194  if (!candidates.missing(p))
195  roots.appendStrip(p);
196  }
197 
198  result = MultiIndexSet(roots);
199  total += result;
200  }
201 
202  if (total.empty()) return MultiIndexSet(); // current was empty and no roots could be added
203 
204  int max_kids = rule->getMaxNumKids();
205  int max_relatives = rule->getMaxNumParents() + max_kids;
206  Data2D<int> update;
207  do{
208  update = Data2D<int>(num_dimensions, 0);
209 
210  for(int i=0; i<total.getNumIndexes(); i++){
211  std::vector<int> relative(total.getIndex(i), total.getIndex(i) + num_dimensions);
212  for(auto &r : relative){
213  int k = r; // save the value
214  for(int j=0; j<max_relatives; j++){
215  r = (j < max_kids) ? rule->getKid(k, j) : ((j - max_kids == 0) ? rule->getParent(k) : rule->getStepParent(k));
216  if ((r != -1) && !candidates.missing(relative) && total.missing(relative))
217  update.appendStrip(relative);
218  }
219  r = k;
220  }
221  }
222 
223  if (update.getNumStrips() > 0){
224  MultiIndexSet update_set(update);
225  result += update_set;
226  total += update_set;
227  }
228  }while(update.getNumStrips() > 0);
229 
230  return result;
231 }

241 std::vector<Data2D<T>> splitByLevels(size_t stride, typename std::vector<T>::const_iterator ibegin, typename std::vector<T>::const_iterator iend, std::vector<int>::const_iterator ilevels){
242  size_t top_level = (size_t) *std::max_element(ilevels, ilevels + std::distance(ibegin, iend) / stride);
243 
244  std::vector<Data2D<T>> split(top_level + 1, Data2D<T>(stride, 0));
245 
246  for( struct { std::vector<int>::const_iterator il; typename std::vector<T>::const_iterator idata;} v = {ilevels, ibegin};
247  v.idata != iend;
248  v.il++, std::advance(v.idata, stride))
249  split[*v.il].appendStrip(v.idata);
250 
251  return split;
252 }
