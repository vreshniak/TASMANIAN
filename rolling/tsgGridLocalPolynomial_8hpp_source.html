173  template<int mode>
174  void walkTree(const MultiIndexSet &work, const double x[], std::vector<int> &sindx, std::vector<double> &svals, double *y) const{
175  std::vector<int> monkey_count(top_level+1); // traverse the tree, counts the branches of the current node
176  std::vector<int> monkey_tail(top_level+1); // traverse the tree, keeps track of the previous node (history)
177 
178  for(const auto &r : roots){
179  bool isSupported;
180  double basis_value = evalBasisSupported(work.getIndex(r), x, isSupported);
181 
182  if (isSupported){
183  if (mode == 0){
184  double const *s = surpluses.getStrip(r);
185  for(int k=0; k<num_outputs; k++) y[k] += basis_value * s[k];
186  }else{
187  sindx.push_back(r);
188  svals.push_back(basis_value);
189  }
190 
191  int current = 0;
192  monkey_tail[0] = r;
193  monkey_count[0] = pntr[r];
194 
195  while(monkey_count[0] < pntr[monkey_tail[0]+1]){
196  if (monkey_count[current] < pntr[monkey_tail[current]+1]){
197  int p = indx[monkey_count[current]];
198  basis_value = evalBasisSupported(work.getIndex(p), x, isSupported);
199  if (isSupported){
200  if (mode == 0){
201  double const *s = surpluses.getStrip(p);
202  for(int k=0; k<num_outputs; k++) y[k] += basis_value * s[k];
203  }else{
204  sindx.push_back(p);
205  svals.push_back(basis_value);
206  }
207 
208  monkey_tail[++current] = p;
209  monkey_count[current] = pntr[p];
210  }else{
211  monkey_count[current]++;
212  }
213  }else{
214  monkey_count[--current]++;
215  }
216  }
217  }
218  }

220  // according to https://docs.nvidia.com/cuda/cusparse/index.html#sparse-format
221  // "... it is assumed that the indices are provided in increasing order and that each index appears only once."
222  // This may not be a requirement for cusparseDgemvi(), but it may be that I have not tested it sufficiently
223  // Also, see AccelerationDataGPUFull::cusparseMatveci() for inaccuracies in Nvidia documentation
224  if (mode == 2){
225  std::vector<int> map(sindx);
226  std::iota(map.begin(), map.end(), 0);
227  std::sort(map.begin(), map.end(), [&](int a, int b)->bool{ return (sindx[a] < sindx[b]); });
228 
229  std::vector<int> idx = sindx;
230  std::vector<double> vls = svals;
231  std::transform(map.begin(), map.end(), sindx.begin(), [&](int i)->int{ return idx[i]; });
232  std::transform(map.begin(), map.end(), svals.begin(), [&](int i)->double{ return vls[i]; });
233  }
234  }

267  // synchronize with tasgpu_devalpwpoly_feval
268  template<int ord, TypeOneDRule crule, typename T>
269  Data2D<T> encodeSupportForGPU(const MultiIndexSet &work) const{
270  Data2D<T> cpu_support(num_dimensions, work.getNumIndexes());
271  for(int i=0; i<work.getNumIndexes(); i++){
272  const int* p = work.getIndex(i);
273  T *s = cpu_support.getStrip(i);
274  for(int j=0; j<num_dimensions; j++){
275  s[j] = static_cast<T>(rule->getSupport(p[j]));
276  if (ord != 0){
277  if (ord == 2) s[j] *= s[j];
278  if ((crule == rule_localp) || (crule == rule_semilocalp)) if (p[j] == 0) s[j] = static_cast<T>(-1.0); // constant function
279  if ((crule == rule_localp) && (ord == 2)){
280  if (p[j] == 1) s[j] = static_cast<T>(-2.0);
281  else if (p[j] == 2) s[j] = static_cast<T>(-3.0);
282  }
283  if ((crule == rule_semilocalp) && (ord == 2)){
284  if (p[j] == 1) s[j] = static_cast<T>(-4.0);
285  else if (p[j] == 2) s[j] = static_cast<T>(-5.0);
286  }
287  if ((crule == rule_localpb) && (ord == 2)){
288  if (p[j] < 2) s[j] = static_cast<T>(-2.0); // linear functions on level 0
289  }
290  }
291  }
292  }
293  return cpu_support;
294  }
