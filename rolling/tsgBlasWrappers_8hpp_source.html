318  template<typename scalar_type>
319  inline void solveLS(char trans, int N, int M, scalar_type A[], scalar_type b[], int nrhs = 1){
320  std::vector<scalar_type> work(1);
321  int n = (trans == 'N') ? N : M;
322  int m = (trans == 'N') ? M : N;
323  char effective_trans = (trans == 'N') ? trans : get_trans(static_cast<scalar_type>(0.0));
324  conj_matrix(N, M, A); // does nothing in the real case, computes the conjugate in the complex one
325  TasBLAS::gels(effective_trans, n, m, nrhs, A, n, b, N, work.data(), -1);
326  work.resize(static_cast<size_t>(std::real(work[0])));
327  TasBLAS::gels(effective_trans, n, m, nrhs, A, n, b, N, work.data(), static_cast<int>(work.size()));
328  }
336  template<typename scalar_type>
337  inline void factorizeLQ(int rows, int cols, scalar_type A[], std::vector<scalar_type> &T){
338  T.resize(5);
339  std::vector<scalar_type> work(1);
340  geql(rows, cols, A, rows, T.data(), -1, work.data(), -1);
341  T.resize(static_cast<size_t>(std::real(T[0])));
342  work.resize(static_cast<size_t>(std::real(work[0])));
343  geql(rows, cols, A, rows, T.data(), static_cast<int>(T.size()), work.data(), static_cast<int>(work.size()));
344  }
